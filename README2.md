#include <stdio.h>

int main()
{
    int i,n,j,k=1;
    printf("Enter the number of rows: ");
    scanf("%d",&n);
    for(i=0;i<=n-1;i++){
        for(j=1;j<=(2*i)+1;j++){
            printf("%d",k);
            
        }
        k=k+1;
        printf("\n");
        
    }

    return 0;
}
